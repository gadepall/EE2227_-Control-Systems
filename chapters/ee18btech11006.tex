\item
\begin{align}
 \dot{x(t)}=Ax(t)+Bu(t) \\
 y(t)=Cx(t)+Du(t)
\end{align} 
Taking Laplace transform on both sides we have the following equations
\begin{align}
sIX(s)= AX(s)+ BU(s)\\
(sI-A)X(s)= BU(s)\\
X(s)={(sI-A)^{-1}}B U(s)
\end{align}
and
\begin{align}
Y(s)= CX(s)+DIU(s)
\end{align}
substituting 0.1.1 in this equation gives
\begin{align}
Y(s)=( C{(sI-A)^{-1}}B+DI) U(s)
\end{align}
\begin{align}
H(s)= {\frac{Y(s)}{U(s)}}= C{(sI-A)^{-1}}B+DI
\end{align}
In the above question D=0, so our transfer function becomes
\begin{align}
C{(sI-A)^{-1}}B=\frac{1}{s^3 + 3s^2+2s + 1}
\end{align}

Now, 
\begin{align} 
{\frac{Y(s)}{U(s)}}= {\frac{\frac{Y(s)}{X_{1}(s)}}{\frac{U(s)}{X_{1}(s)}}}  
\end{align}
We get
\begin{align}
 {\frac{Y(s)}{X_{1}(s)}}= 1
\end{align}
and
\begin{align}
{\frac{U(s)}{X_{1}(s)}}={s^3 + 3s^2+2s + 1}
\end{align}
